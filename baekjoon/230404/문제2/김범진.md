```python
import sys
N, M = map(int, sys.stdin.readline().split())
S = set(map(int, sys.stdin.readline().split()))

s = set([i for i in range(1, 1002)]) - S

if (1 in s) and (N in s):
    answer = 0
else:
    min_ = abs(N-1001**3)
    for i in s:
        for j in s:
            for k in s:
                q = i*j*k
                if abs(N-q) < min_:
                    min_ = abs(N-q)
                if (N+1 < q): 
                    break
        if min_ == 0:
            break
    answer = min_

print(answer)
```
